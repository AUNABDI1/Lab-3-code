#include <iostream>
#include <iomanip>
#include <string>
using namespace std;

int main() {
    string name1, name2;
    int qty1, qty2;
    double price1, price2;

    cout << "Product 1 Name: "<< endl;
    cin >> name1;
    cout << "Quantity: ";
    cin >> qty1;
    cout << "Price: ";
    cin >> price1;
    

    cout << "\nProduct 2 Name: "<< endl;
    cin >> name2;
    cout << "Quantity: ";

    cout << "Price: ";
    cin >> price2;

    double total1 = qty1 * price1;
